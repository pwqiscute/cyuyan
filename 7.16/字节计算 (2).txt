int main() {
	int a[5][5];
	int(*p)[4];
	p =(int(*)[4]) a;
	printf("%p,%d", &p[4][2] - &a[4][2], &p[4][2] - &a[4][2]);
}
//------------p是一个数组指针指向一个为4个int类型的数组；把a的第一行地址强行付给p，p指向a的地址；p+1=跳过一个p类型，就是往后跳4个int类型，p【4】表示跳了4次，后面的【2】表是跳了4次后往后数第2个的int类型的数-a【3】【3】；指针的减法是2指针之间的元素个数-（-4），（-4）；因为地址没有负数，（-4）的补码就是其内存地址；
