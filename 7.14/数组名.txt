2.&数组名，数组名表示的是整个数组，&数组名取出的是整个数组的地址；int main() {
	int arr1[] = { 1,2,3,4,5 };
	int arr2[] = { 1,3,4,5,6 };
	int arr3[] = { 2,4,5,6,8 };
	int* arr4[] = { arr1,arr2,arr3 };
	for (int i = 0; i < 3; i++) {
		for (int j = 0; j < 5; j++) {
			printf("%d", *(arr4[i] + j));
		}
		printf("\n");
	}
}
